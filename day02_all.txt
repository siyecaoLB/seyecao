1.yum源管理使用
	linux = 内核 + 周边应用[包]
	/etc/yum.repos.d/xx.repo 软件源配置文件
	[root@room9pc01 yum.repos.d]# cat CentOS.repo
		[CentOS7]   ==> 标题/名称
		name=CentOS 7  ==> 描述信息
		baseurl=file:///var/ftp/centos-1804
		enabled=1 ==> 启用，1启用，0 不启用
		gpgcheck=0 ==> 
			是否检测完整性，0不检测，1检测
	baseurl:
		file:// 本地
		http:// 网站
		ftp://  ftp
	file:// 本地 /var/ftp/centos-1804
	yum : 管理软件包
		yum list ： 列出所有的软件包
		yum repolist : 总共有多少软件包
		yum install 包名 [包2...] :
						安装一个软件包
		]# yum install httpd
		]# yum -y install httpd
		yum remove 包名: 删除一个软件包
	rpm : 查看包有没有装
		rpm -qa 包名 : 
				如果安装了，就显示完整包名
		rpm -ql 包名：
				安装这个包，产生的新的文件
